# OPENCL_EXAMPLE
OpenCL simple examples. Tested on Ubuntu 18.04.

# Installation

$ sudo apt-get install ocl-icd-opencl-dev ocl-icd-libopencl1 opencl-headers ocl-icd-dev clinfo beignet

$ sudo apt install libopencv-dev

# Build

$ cd opencl_example

$ mkdir build

$ cd build

$ cmake ../

$ make

# To run the code:

$ cd ../build/bin

$ ./openclc
